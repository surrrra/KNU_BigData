def get_income(large_char):
    cur=get_cursor()
    
    if large_char=='가구형태별':
        sql="""
        select 대특성, 소특성, avg(근로소득) 근로소득, avg(사업소득) 사업소득,
        avg(재산소득) 재산소득, avg(사적이전소득) 사적이전소득, avg(공적이전소득), avg(사적이전소득) from income
        group by 대특성, 소특성
        having 대특성=%s and 소특성 in ('노인독거', '노인부부', '자녀동거')
        order by 1, 2
        """
        
        cur.execute(sql, large_char)
        rows=cur.fetchall()
        income_df[large_char]=pd.DataFrame(rows, columns=['대특성', '소특성', '근로소득', '사업소득', '재산소득', '사적이전소득', '공적이전소득', '사적연금소득'])
        income_df[large_char]=income_df[large_char].set_index(['대특성', '소특성'])
        
    else:
        sql="""
        select 대특성, 소특성, avg(근로소득) 근로소득, avg(사업소득) 사업소득,
        avg(재산소득) 재산소득, avg(사적이전소득) 사적이전소득, 
        avg(공적이전소득) 공적이전소득, avg(사적연금소득) 사적연금소득 from income
        group by 대특성, 소특성
        having 대특성=%s
        order by 1, 2
        """
        cur.execute(sql, large_char)
        rows=cur.fetchall()
        income_df[large_char]=pd.DataFrame(rows, columns=['대특성', '소특성', '근로소득', '사업소득', '재산소득', '사적이전소득', '공적이전소득', '사적연금소득'])
        income_df[large_char]=income_df[large_char].set_index(['대특성', '소특성'])
        
    cur.close()

연령이 높아질수록 근로, 사업소득이 현저하게 낮아짐. => 75세 이상의 노년층에게 일할 기회 필요

def get_burden(large_char):
    cur=get_cursor()
    
    if large_char=='가구형태별':
        sql="""
        select 대특성, 소특성, avg(식비) as 식비, avg(주거관련비) 주거관련비, avg(보건의료비) 보건의료비,
        avg(월세) 월세, avg(경조사비) 경조사비 from burden
        group by 대특성, 소특성
        having 대특성=%s and 소특성!='기타'
        order by 1, 2
        """
        
    else:
        sql="""
        select 대특성, 소특성, avg(식비) as 식비, avg(주거관련비) 주거관련비, avg(보건의료비) 보건의료비,
        avg(월세) 월세, avg(경조사비) 경조사비 from burden
        group by 대특성, 소특성
        having 대특성=%s
        order by 1, 2
        """
        
    cur.execute(sql, large_char)
    rows=cur.fetchall()
    burden_df[large_char]=pd.DataFrame(rows, columns=['대특성', '소특성', '식비', '주거관련비', '보건의료비', '월세', '경조사비'])
    burden_df[large_char]=burden_df[large_char].set_index(['대특성', '소특성'])
    burden_df[large_char]=burden_df[large_char].astype('float')
    
    cur.close()
